
层序遍历，在最后一个vec取第一个元素即可

![[Pasted image 20241125123747.png|200]]

```cpp
//BFS这棵二叉树,先把右儿子入队,再把左儿子入队.最后一个出队的节点就是左下角的节点
class Solution {
public:
    int findBottomLeftValue(TreeNode *root) {
        TreeNode *node;
        queue<TreeNode *> q;
        q.push(root);
        while (!q.empty()) {
            node = q.front(); q.pop();
            if (node->right) q.push(node->right);
            if (node->left)  q.push(node->left);
        }
        return node->val;
    }
};
```
