
```cpp
const int MAXV = 100;   // 图中最多的顶点数
const int INF = 0x3f3f3f3f; // 用 INF 表示 ∞

class MatGraph {
public:
    int edges[MAXV][MAXV]; // 邻接矩阵数组，假设元素 aij 类型为 int
    int n, e;             // 顶点数和边数
    string vexs[MAXV];    // 存放顶点信息
};

```

```cpp
void CreateMatGraph(int a[][MAXV], int n, int e) {//创建邻接矩阵
    this->n = n; this->e = e; // 设置顶点数和边数
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            this->edges[i][j] = a[i][j];
}
```
