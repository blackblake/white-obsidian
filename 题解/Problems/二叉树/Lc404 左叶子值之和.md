int sumOfLeftLeaves(TreeNode* root) {

        int n=0;

        if(root){
			//如果root的左孩子存在,且是叶子节点,则加上它的值
            if(root->left&&root->left->left==nullptr&&root->left->right==nullptr)
             {n+=root->left->val;}

            n+=sumOfLeftLeaves(root->left);//左右递归
            n+=sumOfLeftLeaves(root->right);

            return n;

        }else{//如果root=null,返回0
            return 0;
        }

    }
