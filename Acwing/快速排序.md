
分治算法：最怕的是“无限划分”!

```cpp
void quick_sort(int q[], int l, int r) {

    // 递归结束条件
    if (l >= r) {
        return;
    }

    int i = l, j = r; int x = q[(l + r) >> 1]; // 选择中间元素作为基准

    // 执行分区操作
    while (i <= j) {
        while (q[i] < x) {
            i++;
        }
        while (q[j] > x) {
            j--;
        }
        if (i <= j) {
            swap(q[i], q[j]);
            i++;
            j--;
        }
    }

    // 递归左右子数组
    quick_sort(q, l, j);
    quick_sort(q, i, r);
}
```
